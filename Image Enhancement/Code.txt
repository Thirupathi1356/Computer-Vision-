import cv2
import numpy as np
import matplotlib.pyplot as plt

# 1. Load the input image
image_path = 'C:\Pictures\Screenshots\Screenshot (40).jpeg'  # REPLACE WITH YOUR ACTUAL IMAGE NAME
image_bgr = cv2.imread(image_path)

if image_bgr is None:
    print(f"Error: Could not find '{image_path}'. Check the path or upload the file.")
else:
    # --- STEP 1: ENHANCE CLARITY (CONTRAST) ---
    # Convert from BGR to LAB color space
    # LAB separates the Lightness (L) from the Colors (A and B)
    lab = cv2.cvtColor(image_bgr, cv2.COLOR_BGR2LAB)
    l_channel, a_channel, b_channel = cv2.split(lab)
    
    # Apply CLAHE to the L-channel (Lightness) only
    # This enhances details and shadows without messing up the actual colors
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    cl = clahe.apply(l_channel)
    
    # Merge the enhanced L-channel back with the original A and B channels
    merged_lab = cv2.merge((cl, a_channel, b_channel))
    clarity_enhanced_bgr = cv2.cvtColor(merged_lab, cv2.COLOR_LAB2BGR)

    # --- STEP 2: ENHANCE COLOR (SATURATION) ---
    # Convert to HSV (Hue, Saturation, Value) color space
    hsv = cv2.cvtColor(clarity_enhanced_bgr, cv2.COLOR_BGR2HSV)
    h, s, v = cv2.split(hsv)
    
    # Multiply the Saturation channel to make colors pop (e.g., 1.2 = 20% boost)
    # We use cv2.multiply to prevent numbers from wrapping around past 255
    s = cv2.multiply(s, 1.3)
    s = np.clip(s, 0, 255).astype(np.uint8) 
    
    # Merge back together
    final_hsv = cv2.merge((h, s, v))
    
    # Convert to RGB for displaying correctly in Matplotlib
    final_rgb = cv2.cvtColor(final_hsv, cv2.COLOR_HSV2RGB)
    original_rgb = cv2.cvtColor(image_bgr, cv2.COLOR_BGR2RGB)

    # --- STEP 3: DISPLAY BEFORE AND AFTER ---
    plt.figure(figsize=(15, 7)) # Adjust size as needed
    
    # Show Original
    plt.subplot(1, 2, 1)
    plt.imshow(original_rgb)
    plt.title('Original Image')
    plt.axis('off')
    
    # Show Enhanced
    plt.subplot(1, 2, 2)
    plt.imshow(final_rgb)
    plt.title('Enhanced Color & Clarity')
    plt.axis('off')
    
    plt.tight_layout()
    plt.show()
